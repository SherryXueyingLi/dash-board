After init, add signal dash board by using function .addBoard(): 

dashboard.addBoard({
		title:'Init the board',
		minHeight: 100,
		content: 'Some HTML'
	}).addBoard({
		title:'Init Board Options',
		minHeight: 300,
		column: 1,
	}).addBoard({
		title:'Add Single Board',
		minHeight: 120,
		column: 2,
		contentUrl: 'src/content/addboard.html'
	})

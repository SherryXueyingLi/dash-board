var onloadFunc = function(){
    var div = document.getElementById('ko');
    require(["knockout", "vm"],function(ko, vm){
        ko.applyBindings(vm, div);
    });
}

dashboard.addBoard({
    title: 'knockout Binding',
    minHeight: 180,
    column: 3,
    contentUrl: 'src/content/knockout.html',
    onLoad: onloadFunc
});
